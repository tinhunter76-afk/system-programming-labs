# 1. Запуск процесса sleep в фоновом режиме
sleep 300 &

# 2. Поиск PID процесса sleep
ps aux | grep sleep

# 3. Просмотр информации о процессе через /proc
cat /proc/[PID]/status

# 4. Просмотр процесса в top (интерактивно)
top

# 5. Просмотр открытых файлов процесса
lsof -p [PID]

# 6. Отправка сигнала SIGTERM процессу
kill [PID]

# 7. Проверка, что процесс завершён
ps aux | grep sleep
